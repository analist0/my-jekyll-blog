type ApiResponse<T> = {

  loading: boolean;
  error?: string;
};

export function useApi<T>(url: string): ApiResponse<T> {
  const [state, setState] = useState<ApiResponse<T>>({ data: {} as T, loading: true });
  
  useEffect(() => {
    fetch(url)
      .then(res => res.json())
      .then(data => setState({ data, loading: false }))

  }, [url]);
  
  return state;
}
