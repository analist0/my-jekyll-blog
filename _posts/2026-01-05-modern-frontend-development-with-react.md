const socket = io('http://localhost:3001');

function Chat() {
  const [messages, setMessages] = useState([]);

      <ul>{messages.map(msg => <li key={msg.id}>{msg.text}</li>)}</ul>
      <input value={input} onChange={e => setInput(e.target.value)} />
