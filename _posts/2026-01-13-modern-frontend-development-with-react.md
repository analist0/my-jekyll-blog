  const toggleTheme = () => setTheme(prev => prev === 'light' ? 'dark' : 'light');

    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
};

  if (!context) throw new Error('useTheme must be used within ThemeProvider');

};
