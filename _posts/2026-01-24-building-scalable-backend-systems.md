**×”×¡×‘×¨**: FastAPI ×ª×•××š ×‘-Async, Pydantic ×œ-Validation. × ×™×¦×•×¨ endpoint ×œ-users.

```python
# app.py - Basic FastAPI Server
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import List
import uvicorn

app = FastAPI(title="Scalable Backend Demo")

class User(BaseModel):
    id: int
    name: str
    email: str

users_db = []  # In-memory DB for demo

@app.post("/users/")
async def create_user(user: User):
    """Create a new user"""
    if any(u.id == user.id for u in users_db):
        raise HTTPException(status_code=400, detail="User ID exists")
    users_db.append(user)
    return user

@app.get("/users/{user_id}")
async def get_user(user_id: int):
    """Get user by ID"""
    user = next((u for u in users_db if u.id == user_id), None)
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    return user

@app.get("/users/")
async def list_users():
    """List all users"""
    return users_db

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

const express = require('express');
const app = express();
app.use(express.json());

let users = []; // In-memory DB

app.post('/users/', (req, res) => {
  const user = req.body;
  if (users.find(u => u.id === user.id)) {
    return res.status(400).json({ error: 'User ID exists' });
  }
  users.push(user);
  res.json(user);
});

app.get('/users/:id', (req, res) => {
  const user = users.find(u => u.id === parseInt(req.params.id));
  if (!user) return res.status(404).json({ error: 'User not found' });
  res.json(user);
});

app.get('/users/', (req, res) => res.json(users));

app.listen(3000, () => console.log('Server on port 3000'));
